class Liquid
  def drink
    puts "You drank some liquid"
  end
  def evaporate
    puts "There is no more liquid"
  end
end

class Alcohol < Liquid
  def drink
    puts "You are now drunk"
  end
end

```ruby
class Liquid
  def drink
    puts "You drank some liquid"
  end
  def evaporate
    puts "There is no more liquid"
  end
end

class Alcohol < Liquid
  def drink
    super
    puts "You are now drunk"
  end
end
```
